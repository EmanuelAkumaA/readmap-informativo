HTML TAGS

<head>  --> tag de cabeça do html, onde ficarão link de estilização, titulo para a pagina e o icone da página
<title> --> titulo da pagina
<link rel="stylesheet" href="style.css"> --> link para o arquivo css
<link rel="icon" href="imagens/favicon.png"> --> link para o icone da pagina
<body>  --> tag body, onde ficarão os elementos da pagina
<h1> --> titulo principal
<h2> --> titulo secundario
<h3> --> titulo terciario
<h4> --> titulo quaternario
<h5> --> titulo quinario
<h6> --> titulo sexundoario
<p> --> paragrafo
<br> --> linha nova
<hr />  -- > serve para colocar uma linha separadora entre os elementos, linha como: _____________________
<img src="imagens/imagem.png" alt="imagem"> --> link para a imagem
<a href="https://www.google.com.br/" target="_blank"> --> link para o google
<ul>  --> Lista não ordenada, com o item 1, item 2, item 3 e item 4
<li> --> item da lista
<ol>  --> Lista ordenada, com o item 1, item 2, item 3 e item 4
<li> --> item da lista
<table border="1"> --> tabela
<tr> --> linha da tabela
<td> --> celula da tabela
<form> --> formulario
<input type="text" name="nome" id="nome" placeholder="nome"> --> campo de entrada
<input type="email" name="email" id="email" placeholder="email"> --> campo de entrada
<input type="passwrd" name="senha" id="senha" placeholder="senha"> --> campo de entrada
<small>texto pequeno</small> --> serve para criar um texto pequeno.
<strong>texto em negrito</strong> --> serve para criar um texto em negrito.
<i>texto em itálico</i> --> serve para criar um texto em itálico.
<mark>texto marcado</mark> --> serve para criar um texto marcado.
<em>texto enfatizado</em> --> serve para criar um texto enfatizado.
